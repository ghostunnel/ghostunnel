{{- /* Previous/next navigation across all docs pages, in sidebar order. */ -}} {{- $pages := slice -}} {{- range (.Site.GetPage "/docs").Sections.ByWeight -}} {{- range .RegularPages.ByWeight -}} {{- $pages = $pages | append . -}} {{- end -}} {{- end -}} {{- $current := . -}} {{- $idx := -1 -}} {{- range $i, $p := $pages -}} {{- if eq $p.RelPermalink $current.RelPermalink -}}{{- $idx = $i -}}{{- end -}} {{- end -}} {{- if ge $idx 0 -}}
{{- if gt $idx 0 -}} {{- $prev := index $pages (sub $idx 1) }} ← Previous {{ $prev.Title }} {{- else }} {{- end }} {{- if lt $idx (sub (len $pages) 1) -}} {{- $next := index $pages (add $idx 1) }} Next → {{ $next.Title }} {{- else }} {{- end }}
{{- end -}}
